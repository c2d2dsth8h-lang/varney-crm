Google Sheets Backup
Requires desktop browser for initial setup. One time only.
Step 1
Go to sheets.google.com on desktop. Create a blank spreadsheet named Varney CRM Backup.
Step 2
Click Extensions → Apps Script. Delete existing code, paste the script below, click Save.
function doPost(e){try{var s=SpreadsheetApp.getActiveSpreadsheet(),data=JSON.parse(e.postData.contents),cl=data.clients||[];var cs=s.getSheetByName("Clients")||s.insertSheet("Clients");cs.clearContents();cs.appendRow(["ID","Business","Owner","Industry","Stage","Phone","Email","Website","Facebook","Fee","AdSpend","JobValue","Intel","Angle","Notes","Created"]);cl.forEach(function(c){cs.appendRow([c.id,c.bizName,c.owner,c.industry,c.stage,c.phone,c.email,c.website,c.facebook,c.fee,c.adSpend,c.jobValue,c.intel,c.angle,c.notes,new Date(c.createdAt)]);});var ds=s.getSheetByName("Discovery")||s.insertSheet("Discovery");ds.clearContents();ds.appendRow(["Business","Owner","Q1","Q2","Q3","Q4","Q5","Q6","Q7","Q8","Q9","Q10"]);cl.forEach(function(c){var d=c.discovery||[];ds.appendRow([c.bizName,c.owner,d[0],d[1],d[2],d[3],d[4],d[5],d[6],d[7],d[8],d[9]]);});var ors=s.getSheetByName("Outreach")||s.insertSheet("Outreach");ors.clearContents();ors.appendRow(["Business","Type","Message","Result","Date"]);cl.forEach(function(c){(c.outreach||[]).forEach(function(o){ors.appendRow([c.bizName,o.type,o.body,o.result,new Date(o.date)]);});});return ContentService.createTextOutput(JSON.stringify({status:"ok"})).setMimeType(ContentService.MimeType.JSON);}catch(err){return ContentService.createTextOutput(JSON.stringify({status:"error",msg:err.toString()})).setMimeType(ContentService.MimeType.JSON);}}
Step 3
Click Deploy → New deployment. Type: Web app. Execute as: Me. Access: Anyone. Deploy, authorize, copy the Web App URL.
Step 4
Paste Web App URL below and tap Save & Test.
